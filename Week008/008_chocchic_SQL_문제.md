```sql  
-- 경기도에 위치한 식품창고 목록 출력하기
SELECT WAREHOUSE_ID, WAREHOUSE_NAME,	ADDRESS,	
    CASE WHEN FREEZER_YN IS NULL THEN 'N'
    ELSE FREEZER_YN 
    END AS FREEZER_YN
FROM FOOD_WAREHOUSE
WHERE ADDRESS LIKE '경기%'
ORDER BY WAREHOUSE_ID ASC
```  

```sql
-- 이름이 없는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NULL
ORDER BY ANIMAL_ID ASC
```  

``` sql
-- 이름이 있는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS
WHERE NAME IS NOT NULL
ORDER BY ANIMAL_ID ASC
```  

```sql
-- NULL 처리하기
SELECT ANIMAL_TYPE,	
    CASE WHEN NAME IS NULL THEN 'No name'
    ELSE NAME
    END AS NAME,
    SEX_UPON_INTAKE
FROM ANIMAL_INS 
ORDER BY ANIMAL_ID ASC
```  

```sql
-- 나이 정보가 없는 회원 수 구하기
SELECT COUNT(*)
FROM USER_INFO
WHERE AGE IS NULL
```  

```sql
-- 평균 일일 대여 요금 구하기
SELECT ROUND(AVG(DAILY_FEE), 0) AS AVERAGE_FEE 
FROM CAR_RENTAL_COMPANY_CAR 
WHERE CAR_TYPE = 'SUV'
```  

```sql
-- 최솟값 구하기
SELECT MIN(DATETIME)
FROM ANIMAL_INS
```  

``` sql
-- 가격이 제일 비싼 식품의 정보 출력하기
SELECT PRODUCT_ID,	PRODUCT_NAME,	PRODUCT_CD,	CATEGORY,	PRICE
FROM FOOD_PRODUCT
WHERE PRICE = (SELECT MAX(PRICE) FROM FOOD_PRODUCT)

-- 다른 방법
SELECT * FROM (
SELECT PRODUCT_ID,	PRODUCT_NAME,	PRODUCT_CD,	CATEGORY,	PRICE
FROM FOOD_PRODUCT
ORDER BY TO_NUMBER(PRICE) DESC
)
WHERE ROWNUM = 1
```
